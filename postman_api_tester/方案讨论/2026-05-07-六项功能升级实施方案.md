## 升级三：响应时间记录与展示

### 目标
在每个接口请求结果中记录 HTTP 响应时间（毫秒），并在报告视图中展示单条响应时间及全局统计（平均值、最大值、P95）。

### 代码修改点

#### 1. `postman_api_tester/postman_api_tester.py` — `execute_test()` 方法（约 line 420）

**修改位置**：在 `response = getattr(self.session, method)(url, **request_kwargs)` 这一行前后插入计时逻辑。

```python
# 修改前（约 line 470）：
response = getattr(self.session, method)(url, **request_kwargs)

# 修改后：
import time as _time
_t0 = _time.monotonic()
response = getattr(self.session, method)(url, **request_kwargs)
response_time_ms = round((_time.monotonic() - _t0) * 1000)
```

**修改位置**：将 `response_time_ms` 写入所有三条 return 分支（PASSED、FAILED、ERROR）：

```python
# PASSED 分支 return dict 末尾追加：
'response_time_ms': response_time_ms,

# FAILED 分支 return dict 末尾追加：
'response_time_ms': response_time_ms,

# ERROR 分支（except 块）：response_time_ms 可能未定义，使用 0
```

#### 3. `report_server.py` — 新增 `_build_junit_xml()` 辅助函数

使用标准库 `xml.etree.ElementTree`（无需新依赖）：

```python
import xml.etree.ElementTree as ET

def _build_junit_xml(report: Dict[str, Any]) -> bytes:
    """将报告数据构建为 JUnit XML 格式。"""
    summary = report.get("summary", {})
    collection_name = str(report.get("collection_name") or report.get("report_name", ""))
    results = report.get("results", [])
    
    total = int(summary.get("total", len(results)))
    failed = int(summary.get("failed", 0))
    error = int(summary.get("error", 0))
    duration_str = str(summary.get("duration", "0s"))
    # 解析 duration 为秒数（格式如 "12.34s" 或 "1m23.45s"）
    duration_s = _parse_duration_to_seconds(duration_str)
    
    # 按 folder 分组
    folder_groups: Dict[str, List[Dict]] = {}
    for result in results:
        folder = str(result.get("folder") or collection_name or "default")
        folder_groups.setdefault(folder, []).append(result)
    
    testsuites = ET.Element("testsuites", {
        "name": collection_name,
        "tests": str(total),
        "failures": str(failed),
        "errors": str(error),
        "time": f"{duration_s:.3f}",
    })
    
    for folder, items in folder_groups.items():
        f_total = len(items)
        f_failed = sum(1 for r in items if r.get("status") == "FAILED")
        f_error = sum(1 for r in items if r.get("status") == "ERROR")
        
        testsuite = ET.SubElement(testsuites, "testsuite", {
            "name": folder,
            "tests": str(f_total),
            "failures": str(f_failed),
            "errors": str(f_error),
        })
        
        for result in items:
            rt_ms = float(result.get("response_time_ms", 0) or 0)
            testcase = ET.SubElement(testsuite, "testcase", {
                "name": str(result.get("name", "")),
                "classname": f"{folder}.{str(result.get('method', ''))}",
                "time": f"{rt_ms / 1000:.3f}",
            })
            status = str(result.get("status", ""))
            if status == "FAILED":
                failure = ET.SubElement(testcase, "failure", {
                    "message": str(result.get("message", "")),
                    "type": "AssertionError",
                })
                failure.text = (
                    f"URL: {result.get('url', '')}\n"
                    f"Expected: {result.get('expected_status', 200)}\n"
                    f"Actual: {result.get('status_code', '')}\n"
                    f"Message: {result.get('message', '')}\n"
                    f"ErrCode: {result.get('err_code', '')}"
                )
            elif status == "ERROR":
                error_elem = ET.SubElement(testcase, "error", {
                    "message": str(result.get("message", "")),
                    "type": "RequestError",
                })
                error_elem.text = (
                    f"URL: {result.get('url', '')}\n"
                    f"Message: {result.get('message', '')}"
                )
    
    ET.indent(testsuites, space="  ")  # Python 3.9+
    return b'<?xml version="1.0" encoding="UTF-8"?>\n' + ET.tostring(testsuites, encoding="unicode").encode("utf-8")


def _parse_duration_to_seconds(duration_str: str) -> float:
    """解析 '12.34s' / '1m23.45s' / '2h3m4s' 格式为浮点秒数。"""
    import re as _re
    s = str(duration_str or "0").strip()
    total = 0.0
    for value, unit in _re.findall(r'(\d+\.?\d*)\s*([hms])', s):
        v = float(value)
        if unit == 'h':
            total += v * 3600
        elif unit == 'm':
            total += v * 60
        else:
            total += v
    return total if total > 0 else 0.0
```

> `ET.indent()` 需要 Python 3.9+。若需兼容 3.8，可省略 `ET.indent()` 调用，输出无缩进 XML（对 CI 工具无影响）。

```javascript
let filteredReports = allReports;

function applyListFilter() {
    const q = (document.getElementById('filter-q')?.value || '').toLowerCase().trim();
    const statusMode = document.getElementById('filter-status')?.value || '';
    const minRate = parseFloat(document.getElementById('filter-min-rate')?.value || '0') || 0;
    const fromDate = document.getElementById('filter-from-date')?.value || '';
    const toDate = document.getElementById('filter-to-date')?.value || '';

    filteredReports = allReports.filter(r => {
        // 关键词搜索
        if (q) {
            const text = [r.report_name, r.collection_name, r.source_original_file || ''].join(' ').toLowerCase();
            if (!text.includes(q)) return false;
        }
        // 成功率筛选
        const rate = parseFloat((r.summary?.success_rate || '0').replace('%', '')) || 0;
        if (minRate > 0 && rate < minRate) return false;
        // 状态筛选
        if (statusMode === 'all_passed' && (r.summary?.failed > 0 || r.summary?.error > 0)) return false;
        if (statusMode === 'has_failed' && r.summary?.failed === 0 && r.summary?.error === 0) return false;
        // 日期筛选
        if (fromDate && r.generated_at && r.generated_at.slice(0, 10) < fromDate) return false;
        if (toDate && r.generated_at && r.generated_at.slice(0, 10) > toDate) return false;
        return true;
    });

    renderReportTable(filteredReports);
    const countEl = document.getElementById('filter-count');
    if (countEl) {
        countEl.textContent = filteredReports.length < allReports.length
            ? `筛选结果：${filteredReports.length} / ${allReports.length} 条`
            : '';
    }
}

    document.getElementById('filter-q').value = '';
    document.getElementById('filter-status').value = '';
    document.getElementById('filter-min-rate').value = '';
    document.getElementById('filter-from-date').value = '';
    document.getElementById('filter-to-date').value = '';

}
```

```python
@app.route("/api/retry-failures", methods=["POST"])
def api_retry_failures():
    if not ENABLE_RETRY_FAILURES:
        return jsonify({"error": "一键重试功能未启用"}), 403
    
    data = request.get_json(silent=True) or {}
    report_name = str(data.get("report_name") or "").strip()
    result_indices = data.get("result_indices")  # None = 全部失败；list[int] = 指定
    
    if not report_name:
        return jsonify({"error": "report_name 不能为空"}), 400
    
    try:
        report = find_report(report_name)
    except FileNotFoundError:
        return jsonify({"error": "报告不存在"}), 404
    
    results = report.get("results", [])
    
    # 确定要重试的结果集
    if isinstance(result_indices, list) and result_indices:
        targets = [results[i] for i in result_indices if 0 <= i < len(results) and results[i].get("status") in ("FAILED", "ERROR")]
    else:
        targets = [r for r in results if r.get("status") in ("FAILED", "ERROR")]
    
    if not targets:
        return jsonify({"error": "没有需要重试的失败/错误接口"}), 400
    
    # 读取原始集合文件（从 meta 中获取 source_file）
    source_file = str(report.get("source_file") or "").strip()
    if not source_file or not Path(source_file).exists():
        # 尝试从 uploaded_collections 找到对应 json
        return jsonify({"error": "原始集合文件不存在，无法重试"}), 400
    
    try:
        with open(source_file, encoding="utf-8") as f:
            collection_data = json.load(f)
    except Exception as e:
        return jsonify({"error": f"读取集合文件失败: {e}"}), 500
    
    # 收集 item_path 列表
    selected_paths = []
    for r in targets:
        path = r.get("item_path")
        if isinstance(path, list) and path:
            selected_paths.append(path)
    
    if not selected_paths:
        return jsonify({"error": "失败接口缺少 item_path，无法精确重试"}), 400
    
    # 写入临时集合文件
    job_id = uuid.uuid4().hex
    tmp_collection_path = UPLOADED_COLLECTIONS_DIR / f"{job_id}.json"
    try:
        with tmp_collection_path.open("w", encoding="utf-8") as f:
            json.dump(collection_data, f, ensure_ascii=False)
    except Exception as e:
        return jsonify({"error": f"临时文件写入失败: {e}"}), 500
    
    # 构造重试报告名
    orig_name = str(report.get("collection_name") or report_name).replace(".html", "")
    from datetime import datetime as _dt
    timestamp = _dt.now().strftime("%Y%m%d_%H%M%S")
    retry_report_name = f"{orig_name}_retry_{timestamp}.html"
    
    # 入队执行
    source_original_file = str(report.get("source_file") or "").replace("\\", "/").split("/")[-1]
    job_info = _enqueue_run_job(
        job_id=job_id,
        collection_path=str(tmp_collection_path),
        report_name=retry_report_name,
        source_original_file=f"retry:{source_original_file}",
        selected_item_paths=selected_paths,
    )
    
    return jsonify({"job_id": job_id, "report_name": retry_report_name, "retry_count": len(selected_paths)})
```

#### 4. 新增 `postman_api_tester/assertions.py`

抽取断言引擎为独立模块（避免污染 `postman_api_tester.py`）：

```python
"""JSONPath 断言引擎"""
from typing import Any, Dict, List

SUPPORTED_OPS = {"eq", "ne", "gt", "lt", "gte", "lte", "exists", "not_exists", "contains"}

def evaluate_assertions(response_body: Any, assertions: List[Dict]) -> List[Dict]:
    """
    评估断言规则列表，返回失败断言列表。
    每条断言: {"path": "$.code", "op": "eq", "expected": 0}
    失败记录: {"path": ..., "op": ..., "expected": ..., "actual": ..., "passed": False}
    """
    try:
        from jsonpath_ng import parse as _parse
    except ImportError:
        return [{"path": "*", "op": "*", "expected": "*", "actual": "jsonpath-ng 未安装", "passed": False}]
    
    failures = []
    for rule in assertions:
        if not isinstance(rule, dict):
            continue
        path = str(rule.get("path") or "")
        op = str(rule.get("op") or "eq").lower()
        expected = rule.get("expected")
        
        if not path or op not in SUPPORTED_OPS:
            continue
        
        try:
            expr = _parse(path)
            matches = expr.find(response_body)
            actual_values = [m.value for m in matches]
        except Exception as e:
            failures.append({"path": path, "op": op, "expected": expected, "actual": f"解析错误: {e}", "passed": False})
            continue
        
        if op == "exists":
            passed = len(actual_values) > 0
            actual = len(actual_values) > 0
        elif op == "not_exists":
            passed = len(actual_values) == 0
            actual = len(actual_values) > 0
        else:
            actual = actual_values[0] if actual_values else None
            passed = _compare(actual, op, expected)
        
        if not passed:
            failures.append({"path": path, "op": op, "expected": expected, "actual": actual, "passed": False})
    
    return failures


def _compare(actual: Any, op: str, expected: Any) -> bool:
    try:
        if op == "eq":
            return actual == expected or str(actual) == str(expected)
        elif op == "ne":
            return actual != expected
        elif op == "gt":
            return float(actual) > float(expected)
        elif op == "lt":
            return float(actual) < float(expected)
        elif op == "gte":
            return float(actual) >= float(expected)
        elif op == "lte":
            return float(actual) <= float(expected)
        elif op == "contains":
            return str(expected) in str(actual)
    except (TypeError, ValueError):
        return False
    return False
```
