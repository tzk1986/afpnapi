```python
class PostmanTestException(Exception): pass

# 解析异常（Tier 1：Collection 加载时）
class ParseError(PostmanTestException): pass
    ├─ InvalidCollectionFormat(ParseError)       # JSON 格式错误
    ├─ UnsupportedSpecVersion(ParseError)        # spec 版本不支持
    └─ CollectionStructureError(ParseError)      # 结构不符合预期

# 校验异常（Tier 2：参数验证时）
class ValidationError(PostmanTestException): pass
    ├─ InvalidParameter(ValidationError)         # 参数类型/值不合法
    ├─ MissingAuthToken(ValidationError)         # 认证 token 缺失
    ├─ InvalidUrl(ValidationError)               # URL 格式不合法
    └─ InvalidAssertion(ValidationError)         # 断言配置不合法

# 执行异常（Tier 3：运行时）
class ExecutionError(PostmanTestException): pass
    ├─ RequestTimeout(ExecutionError)            # 请求超时
    ├─ AssertionFailure(ExecutionError)          # 断言执行失败
    ├─ CheckpointRecoveryFailed(ExecutionError)  # 断点恢复失败
    └─ NetworkError(ExecutionError)              # 网络连接错误

# 资源异常（Tier 4：资源管理）
class FileOperationError(PostmanTestException): pass
class CacheError(PostmanTestException): pass
```

#### 异常使用原则

1. **抛出异常时必须指定根因**
   ```python
   # ❌ 不好
   raise ExecutionError("Collection parse failed")
   
   # ✅ 好
   try:
       parse_collection(file)
   except json.JSONDecodeError as e:
       raise InvalidCollectionFormat(f"Invalid JSON at line {e.lineno}: {e.msg}") from e
   ```

2. **错误日志必须包含上下文**
   ```python
   logger.error("parse_collection_failed", extra={
       "collection_file": file_path,
       "error_type": type(e).__name__,
       "error_message": str(e),
   })
   ```

**`core/pipeline.py`（~250 行）**

```python
class ExecutionPipeline:
    """统一的测试执行流程编排。
    
    职责：
    - 管理解析 → 认证 → 执行 → 报告的完整链路
    - 支持断点恢复
    - 不涉及磁盘 I/O（由上层处理）
    """
    
    def __init__(self, config: TestExecutionConfig):
        self.config = config
        self.parser = PostmanApiParser()
        self.executor = PostmanTestExecutor()
        self.checkpoint_mgr = CheckpointManager(config.job_id)
    
    def execute(self, collection_file: str) -> ExecutionResult:
        """完整执行流程，支持断点恢复。
        
        返回值：ExecutionResult(success, results, report_data)
        """
        # 1. 加载检查点（如果有）
        checkpoint = self.checkpoint_mgr.load_if_exists()
        if checkpoint:
            logger.info("Resuming from checkpoint", extra={"job_id": self.config.job_id})
        
        # 2. 解析 Collection
        collection = self.parser.parse(
            collection_file,
            skip_items=checkpoint.executed_items if checkpoint else []
        )
        
        # 3. 执行测试
        results = self.executor.execute(collection)
        
        # 4. 生成报告（不涉及磁盘写入）
        report_data = ReportEngine.generate(results, self.config)
        
        # 5. 清理检查点
        self.checkpoint_mgr.cleanup_after_success()
        
        return ExecutionResult(
            success=all(r.status == "PASSED" for r in results),
            results=results,
            report_data=report_data
        )
```

```python
class BaseHandler:
    """路由处理的基类，统一参数验证与响应包装。
    
    所有 handler 模块中的路由函数应继承或使用这些静态方法。
    """
    
    @staticmethod
    def validate_required_param(
        value: Any,
        param_name: str,
        param_type: type = str
    ) -> Any:
        """统一参数验证。
        
        异常：ValidationError
        """
        if value is None or (isinstance(value, str) and not value.strip()):
            raise ValidationError(f"Missing required parameter: {param_name}")
        if param_type and not isinstance(value, param_type):
            raise ValidationError(
                f"Invalid type for {param_name}: expected {param_type.__name__}, got {type(value).__name__}"
            )
        return value
    
    @staticmethod
    def json_response(
        data: Any,
        status_code: int = 200,
        message: str = "OK"
    ) -> ResponseReturnValue:
        """统一响应包装。"""
        response_body = {
            "code": status_code,
            "message": message,
            "data": data,
            "timestamp": datetime.now().isoformat(),
        }
        return jsonify(response_body), status_code
    
    @staticmethod
    def error_response(
        error: Exception,
        status_code: int = 500
    ) -> ResponseReturnValue:
        """统一错误响应。"""
        logger.error(f"Handler error: {type(error).__name__}: {error}")
        return BaseHandler.json_response(
            {"error": type(error).__name__, "details": str(error)},
            status_code=status_code,
            message="Error"
        )
```

```python
class BaseService:
    """业务逻辑层的基类，统一错误处理与日志。
    
    所有 service 类应继承这个基类。
    """
    
    @staticmethod
    def safe_execute(func: Callable, *args, **kwargs) -> Any:
        """安全执行业务逻辑，统一错误捕获。"""
        try:
            result = func(*args, **kwargs)
            logger.debug(f"Service execution successful: {func.__name__}")
            return result
        except PostmanTestException as e:
            logger.error(f"Service execution failed: {type(e).__name__}: {e}")
            raise
        except Exception as e:
            logger.error(f"Unexpected error in {func.__name__}: {type(e).__name__}: {e}")
            raise ExecutionError(f"Service execution failed: {str(e)}") from e
    
    @staticmethod
    def log_event(
        event_name: str,
        level: str = "info",
        **extra_fields
    ) -> None:
        """结构化日志事件。"""
        getattr(logger, level)(
            event_name,
            extra={**extra_fields, "timestamp": datetime.now().isoformat()}
        )
```
